## Basic Autogenerated Migration

### Scenario: New model added, need to sync database

```bash
# 1. Ensure models are updated
# Edit src/models/new_feature.py

# 2. Run the workflow
python scripts/generate_clean_migration.py

# 3. Review the generated migration
python scripts/clean_autogenerated_migration.py alembic/versions/041_sync_database_with_models.py

# 4. Edit the migration based on analysis
nano alembic/versions/041_sync_database_with_models.py

# 5. Test on clean database
createdb test_clean
export DATABASE_URL="postgresql://user:pass@localhost/test_clean"
alembic upgrade head

# 6. Test on copy of production
pg_dump production_db > backup.sql
createdb test_copy
psql test_copy < backup.sql
export DATABASE_URL="postgresql://user:pass@localhost/test_copy"
alembic upgrade head

# 7. Apply to production
export DATABASE_URL="postgresql://user:pass@localhost/production_db"
pg_dump production_db > pre_migration_backup.sql
alembic upgrade head
```

## Adding New Tables

### Scenario: Adding a new feature with multiple tables

**Models defined:**

```python
# src/models/notifications.py
class NotificationTemplate(Base):
    __tablename__ = 'notification_templates'
    
    id = Column(Integer, primary_key=True)
    institution_id = Column(Integer, ForeignKey('institutions.id'), nullable=False)
    name = Column(String(255), nullable=False)
    template_type = Column(Enum('email', 'sms', 'push', name='notificationtype'), nullable=False)
    subject = Column(String(500), nullable=True)
    body = Column(Text, nullable=False)
    is_active = Column(Boolean, default=True, nullable=False)
    created_at = Column(DateTime, server_default=func.now(), nullable=False)
    updated_at = Column(DateTime, onupdate=func.now(), nullable=True)
```

```python
def upgrade() -> None:
    conn = op.get_bind()
    
    # Create enum type with existence check
    result = conn.execute(
        "SELECT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'notificationtype')"
    ).scalar()
    
    if not result:
        op.execute("CREATE TYPE notificationtype AS ENUM ('email', 'sms', 'push')")
        print("Created enum type: notificationtype")
    else:
        print("Enum type already exists: notificationtype")
    
    # Create table with existence check
    result = conn.execute("""
        SELECT EXISTS (
            SELECT 1 FROM information_schema.tables 
            WHERE table_name = 'notification_templates'
        )
    """).scalar()
    
    if not result:
        op.create_table('notification_templates',
            sa.Column('id', sa.Integer(), nullable=False),
            sa.Column('institution_id', sa.Integer(), nullable=False),
            sa.Column('name', sa.String(length=255), nullable=False),
            sa.Column('template_type', 
                      postgresql.ENUM('email', 'sms', 'push', 
                                     name='notificationtype', 
                                     create_type=False),  # Don't recreate
                      nullable=False),
            sa.Column('subject', sa.String(length=500), nullable=True),
            sa.Column('body', sa.Text(), nullable=False),
            sa.Column('is_active', sa.Boolean(), nullable=False, 
                      server_default='true'),  # Add default
            sa.Column('created_at', sa.DateTime(), nullable=False,
                      server_default=sa.text('now()')),  # Add default
            sa.Column('updated_at', sa.DateTime(), nullable=True),
            sa.PrimaryKeyConstraint('id'),
            sa.ForeignKeyConstraint(['institution_id'], ['institutions.id'], 
                                   ondelete='CASCADE')
        )
        print("Created table: notification_templates")
        
        # Create index
        op.create_index('ix_notification_templates_institution_id', 
                        'notification_templates', ['institution_id'])
        print("Created index: ix_notification_templates_institution_id")
        
        # Enable RLS
        op.execute("ALTER TABLE notification_templates ENABLE ROW LEVEL SECURITY")
        
        # Create RLS policy
        op.execute("""
            CREATE POLICY notification_templates_isolation_policy 
            ON notification_templates
            USING (
                institution_id = current_setting('app.current_institution_id', true)::integer
                OR current_setting('app.bypass_rls', true)::boolean = true
            )
        """)
        print("Created RLS policy: notification_templates_isolation_policy")
    else:
        print("Table already exists: notification_templates")
```

## Modifying Existing Tables

### Scenario: Adding a column to an existing table

**Model change:**

```python
# Add new column to existing model
class User(Base):
    # ... existing columns ...
    email_verified = Column(Boolean, default=False, nullable=False)  # NEW
    email_verified_at = Column(DateTime, nullable=True)  # NEW
```

**Generated migration:**

```python
def upgrade() -> None:
    op.add_column('users', sa.Column('email_verified', sa.Boolean(), nullable=False))
    op.add_column('users', sa.Column('email_verified_at', sa.DateTime(), nullable=True))
```

**Problem:** Existing rows will fail NOT NULL constraint!

**Fixed migration:**

```python
def upgrade() -> None:
    conn = op.get_bind()
    
    # Check if columns exist
    email_verified_exists = conn.execute("""
        SELECT EXISTS (
            SELECT 1 FROM information_schema.columns 
            WHERE table_name = 'users' 
            AND column_name = 'email_verified'
        )
    """).scalar()
    
    if not email_verified_exists:
        # Add as nullable first
        op.add_column('users', sa.Column('email_verified', sa.Boolean(), nullable=True))
        
        # Populate existing rows with default value
        op.execute("UPDATE users SET email_verified = false WHERE email_verified IS NULL")
        
        # Now make it NOT NULL
        op.alter_column('users', 'email_verified', nullable=False, 
                       server_default='false')
        
        print("Added column: users.email_verified")
    else:
        print("Column already exists: users.email_verified")
    
    # Add timestamp column (nullable is fine)
    email_verified_at_exists = conn.execute("""
        SELECT EXISTS (
            SELECT 1 FROM information_schema.columns 
            WHERE table_name = 'users' 
            AND column_name = 'email_verified_at'
        )
    """).scalar()
    
    if not email_verified_at_exists:
        op.add_column('users', sa.Column('email_verified_at', sa.DateTime(), 
                                         nullable=True))
        print("Added column: users.email_verified_at")
    else:
        print("Column already exists: users.email_verified_at")
```

## Working with Enum Types

### Scenario: Shared enum type between multiple tables

**Problem:** Multiple tables use the same enum, autogeneration tries to create it multiple times.

**Models:**

```python
# Multiple models share the same enum
class Assignment(Base):
    status = Column(Enum('draft', 'published', 'archived', 
                        name='contentstatus'), nullable=False)

class Quiz(Base):
    status = Column(Enum('draft', 'published', 'archived', 
                        name='contentstatus'), nullable=False)
```

**Bad generated migration:**

```python
def upgrade() -> None:
    # This will fail if run twice!
    op.execute("CREATE TYPE contentstatus AS ENUM ('draft', 'published', 'archived')")
    
    op.create_table('assignments', ...)
    op.create_table('quizzes', ...)
```

**Fixed migration:**

```python
def upgrade() -> None:
    conn = op.get_bind()
    
    # Create shared enum ONCE with check
    result = conn.execute(
        "SELECT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'contentstatus')"
    ).scalar()
    
    if not result:
        op.execute("CREATE TYPE contentstatus AS ENUM ('draft', 'published', 'archived')")
        print("Created shared enum type: contentstatus")
    else:
        print("Shared enum type already exists: contentstatus")
    
    # Create tables using create_type=False
    table_exists = conn.execute("""
        SELECT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_name = 'assignments')
    """).scalar()
    
    if not table_exists:
        op.create_table('assignments',
            sa.Column('id', sa.Integer(), nullable=False),
            sa.Column('status', 
                      postgresql.ENUM('draft', 'published', 'archived',
                                     name='contentstatus',
                                     create_type=False),  # IMPORTANT: Don't recreate!
                      nullable=False),
            # ... other columns
        )
    
    table_exists = conn.execute("""
        SELECT EXISTS (SELECT 1 FROM information_schema.tables WHERE table_name = 'quizzes')
    """).scalar()
    
    if not table_exists:
        op.create_table('quizzes',
            sa.Column('id', sa.Integer(), nullable=False),
            sa.Column('status', 
                      postgresql.ENUM('draft', 'published', 'archived',
                                     name='contentstatus',
                                     create_type=False),  # IMPORTANT: Don't recreate!
                      nullable=False),
            # ... other columns
        )
```

## Handling Foreign Keys

### Scenario: Adding foreign key with potential orphaned data

**Problem:** Existing data may violate new foreign key constraint.

**Generated migration:**

```python

                          ['teacher_id'], ['id'], ondelete='CASCADE')
```

**Generated migration:**

**Problem:** Existing rows don't have values!

**Fixed migration:**

## Common Patterns

### Pattern 1: Idempotent Table Creation

```python
def upgrade() -> None:
    conn = op.get_bind()
    
    table_name = 'my_table'
    result = conn.execute(
        sa.text(f"""
            SELECT EXISTS (
                SELECT 1 FROM information_schema.tables 
                WHERE table_name = '{table_name}'
            )
        """)
    ).scalar()
    
    if not result:
        op.create_table(table_name, ...)
        print(f"Created table: {table_name}")
    else:
        print(f"Table already exists: {table_name}")
```

### Pattern 2: Idempotent Enum Creation

```python
def upgrade() -> None:
    conn = op.get_bind()
    
    enum_name = 'myenum'
    result = conn.execute(
        sa.text(f"SELECT EXISTS (SELECT 1 FROM pg_type WHERE typname = '{enum_name}')")
    ).scalar()
    
    if not result:
        op.execute(f"CREATE TYPE {enum_name} AS ENUM ('value1', 'value2')")
        print(f"Created enum type: {enum_name}")
    else:
        print(f"Enum type already exists: {enum_name}")
```

### Pattern 3: Safe Column Addition

```python
def upgrade() -> None:
    conn = op.get_bind()
    
    table_name = 'my_table'
    column_name = 'new_column'
    
    result = conn.execute(
        sa.text(f"""
            SELECT EXISTS (
                SELECT 1 FROM information_schema.columns 
                WHERE table_name = '{table_name}' 
                AND column_name = '{column_name}'
            )
        """)
    ).scalar()
    
    if not result:
        op.add_column(table_name, sa.Column(column_name, sa.String(100), nullable=True))
        print(f"Added column: {table_name}.{column_name}")
    else:
        print(f"Column already exists: {table_name}.{column_name}")
```

### Pattern 4: Safe Foreign Key Addition

```python
def upgrade() -> None:
    conn = op.get_bind()
    
    fk_name = 'fk_table_parent'
    result = conn.execute(
        sa.text(f"""
            SELECT EXISTS (
                SELECT 1 FROM information_schema.table_constraints 
                WHERE constraint_name = '{fk_name}'
            )
        """)
    ).scalar()
    
    if not result:
        # Check for orphaned records
        orphans = conn.execute(
            sa.text("""
                SELECT COUNT(*) FROM child_table c
                LEFT JOIN parent_table p ON c.parent_id = p.id
                WHERE c.parent_id IS NOT NULL AND p.id IS NULL
            """)
        ).scalar()
        
        if orphans > 0:
            print(f"WARNING: {orphans} orphaned records - cleaning up")
            # Handle orphaned records appropriately
        
        op.create_foreign_key(fk_name, 'child_table', 'parent_table',
                              ['parent_id'], ['id'], ondelete='CASCADE')
        print(f"Created foreign key: {fk_name}")
    else:
        print(f"Foreign key already exists: {fk_name}")
```

## Testing Checklist

After creating any migration, test:

- [ ] **Clean database test**: All migrations from scratch
- [ ] **Existing database test**: Migration on copy with data
- [ ] **Idempotency test**: Run migration twice (should not error)
- [ ] **Rollback test**: Downgrade then upgrade
- [ ] **Data integrity test**: Verify no data loss
- [ ] **Performance test**: Check migration duration on large tables
- [ ] **Application test**: Run application after migration

## Troubleshooting

### "Type already exists" Error

**Cause:** Enum type being created twice

**Fix:** Add existence check (see Pattern 2)

### "Relation already exists" Error

**Cause:** Table being created twice

**Fix:** Add existence check (see Pattern 1)

### "Violates foreign key constraint" Error

**Cause:** Orphaned records in child table

**Fix:** Clean orphaned data first (see Pattern 4)

### "Cannot add NOT NULL constraint" Error

**Cause:** Existing rows have NULL values

**Fix:** Populate data first (see Data Migrations example)
